60zrtos_task_top_of_stack_t * pxPortInitialiseStack( zrtos_task_top_of_stack_t * pxTopOfStack,
61 zrtos_task_callback_t pxCode,
62 void * pvParameters )
63{
64uint16_t usAddress;
65
66 /* Simulate how the stack would look after a call to vPortYield() generated by
67 * the compiler. */
68
69 /* The start of the task code will be popped off the stack last, so place
70 * it on first. */
71 usAddress = ( uint16_t ) pxCode;
72 *pxTopOfStack = ( zrtos_task_top_of_stack_t ) ( usAddress & ( uint16_t ) 0x00ff );
73 pxTopOfStack--;
74
75 usAddress >>= 8;
76 *pxTopOfStack = ( zrtos_task_top_of_stack_t ) ( usAddress & ( uint16_t ) 0x00ff );
77 pxTopOfStack--;
78
79#if defined(__AVR_3_BYTE_PC__)
80
81 /* The AVR ATmega2560/ATmega2561 have 256KBytes of program memory and a 17-bit
82 * program counter. When a code address is stored on the stack, it takes 3 bytes
83 * instead of 2 for the other ATmega* chips.
84 *
85 * Store 0 as the top byte since we force all task routines to the bottom 128K
86 * of flash. We do this by using the .lowtext label in the linker script.
87 *
88 * In order to do this properly, we would need to get a full 3-byte pointer to
89 * pxCode. That requires a change to GCC. Not likely to happen any time soon.
90 */
91 *pxTopOfStack = 0;
92 pxTopOfStack--;
93#endif
94
95 /* Next simulate the stack as if after a call to portSAVE_CONTEXT().
96 * portSAVE_CONTEXT places the flags on the stack immediately after r0
97 * to ensure the interrupts get disabled as soon as possible, and so ensuring
98 * the stack use is minimal should a context switch interrupt occur. */
99 *pxTopOfStack = ( zrtos_task_top_of_stack_t ) 0x00; /* R0 */
100 pxTopOfStack--;
101 *pxTopOfStack = 0x80;//enable interrupts
102 pxTopOfStack--;
103
104#if defined(__AVR_3_BYTE_PC__)
105
106 /* If we have an ATmega256x, we are also saving the EIND register.
107 * We should default to 0.
108 */
109 *pxTopOfStack = ( zrtos_task_top_of_stack_t ) 0x00; /* EIND */
110 pxTopOfStack--;
111#endif
112
113#if defined(__AVR_HAVE_RAMPZ__)
114
115 /* We are saving the RAMPZ register.
116 * We should default to 0.
117 */
118 *pxTopOfStack = ( zrtos_task_top_of_stack_t ) 0x00; /* RAMPZ */
119 pxTopOfStack--;
120#endif
121
122 /* Now the remaining registers. The compiler expects R1 to be 0. */
123 *pxTopOfStack = ( zrtos_task_top_of_stack_t ) 0x00; /* R1 */
124
125 /* Leave R2 - R23 untouched */
126 pxTopOfStack -= 23;
127
128 /* Place the parameter on the stack in the expected location. */
129 usAddress = ( uint16_t ) pvParameters;
130 *pxTopOfStack = ( zrtos_task_top_of_stack_t ) ( usAddress & ( uint16_t ) 0x00ff );
131 pxTopOfStack--;
132
133 usAddress >>= 8;
134 *pxTopOfStack = ( zrtos_task_top_of_stack_t ) ( usAddress & ( uint16_t ) 0x00ff );
135
136 /* Leave register R26 - R31 untouched */
137 pxTopOfStack -= 7;
138
139 return pxTopOfStack;
140}

283void _zrtos_task_scheduler__isr_start(void){
284 ZRTOS__INTERRUPTS_DISABLE();
285 // Clear registers
286 TCCR1A = 0;
287 TCCR1B = 0;
288 TCNT1 = 0;
289
290 // 1000 Hz (16000000/((249+1)*64))
291 OCR1A = 249;
292 // CTC
293 TCCR1B |= (1 << WGM12);
294 // Prescaler 64
295 TCCR1B |= (1 << CS11) | (1 << CS10);
296 // Output Compare Match A Interrupt Enable
297 TIMSK1 |= (1 << OCIE1A);
298 ZRTOS__INTERRUPTS_ENABLE();
299}

301void _zrtos_task_scheduler__isr_stop(void){
302 ZRTOS__INTERRUPTS_DISABLE();
303 TIMSK1 &= ~(1 << OCIE1A);
304 ZRTOS__INTERRUPTS_ENABLE();
305}
