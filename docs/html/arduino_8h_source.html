13/*
14// the prescaler is set so that timer0 ticks every 64 clock cycles, and the
15// the overflow handler is called every 256 ticks.
16#define MICROSECONDS_PER_TIMER0_OVERFLOW (clockCyclesToMicroseconds(64 * 256))
17// the whole number of milliseconds per timer0 overflow
18#define MILLIS_INC (MICROSECONDS_PER_TIMER0_OVERFLOW / 1000)
19// the fractional number of milliseconds per timer0 overflow. we shift right
20// by three to fit these numbers into a byte. (for the clock speeds we care
21// about - 8 and 16 MHz - this doesn't lose precision.)
22#define FRACT_INC ((MICROSECONDS_PER_TIMER0_OVERFLOW % 1000) >> 3)
23#define FRACT_MAX (1000 >> 3)
24volatile unsigned long timer0_overflow_count = 0;
25volatile unsigned long timer0_millis = 0;
26static unsigned char timer0_fract = 0;
27*/
28
29SIGNAL(TIMER0_OVF_vect){
30 // copy these to local variables so they can be stored in registers
31 // (volatile variables must be read from memory on every access)
32 unsigned long m = timer0_millis;
33 unsigned char f = timer0_fract;
34 m += MILLIS_INC;
35 f += FRACT_INC;
36 if (f >= FRACT_MAX) {
37 f -= FRACT_MAX;
38 m += 1;
39 }
40 timer0_fract = f;
41 timer0_millis = m;
42 timer0_overflow_count++;
43
44 zrtos__on_interrupt(64);
45}
