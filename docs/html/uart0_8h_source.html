116#ifdef URSEL

127 }else{
128 UART0_STATUS &= ~_BV(U2X0);
129 }

131 UBRR0L = (uint8_t) baudrate;

137#ifdef URSEL0

139#else
140 UCSR0C = _BV(UCSZ01)|_BV(UCSZ00);
141#endif
142
143#elif defined(ATMEGA_UART)
144 /* set baud rate */
145 if(ZRTOS_VFS_MODULE_AVR_UART__IS_DOUBLE_SPEED(baudrate)){
146 UART0_STATUS = _BV(U2X); //Enable 2x speed
147 baudrate &= ~0x8000;
