137static bool _zrtos_task_scheduler__switch_task(void){
138 zrtos_task_t *active_task = _zrtos_task_scheduler__get_active_task();
139 zrtos_task_t *task = active_task;
140
141 do{
142 zrtos_task__on_tick(task);
143 task = zrtos_task__get_next_task(task);
144 }while(task != active_task);
145
146 task = active_task;
147 do{
148 task = zrtos_task__get_next_task(task);
149 if(zrtos_task__is_ready(task)){
150 return _zrtos_task_scheduler__set_active_task(task);
151 }
152 }while(task != active_task);
153
154 _zrtos_task_scheduler__set_active_task(
155 _zrtos_task_scheduler__get_sleep_task()
156 );
157
158 return true;
159}
