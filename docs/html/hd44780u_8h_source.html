57 ,thiz->pin_data3, ZRTOS_GPIO_MODE__OUTPUT

59 zrtos_gpio__delay_microseconds(LCD_DELAY_BOOTUP);
60
61 /* initial write to lcd is 8bit */
62 LCD_DATA1_PORT |= _BV(LCD_DATA1_PIN); // LCD_FUNCTION>>4;
63 LCD_DATA0_PORT |= _BV(LCD_DATA0_PIN); // LCD_FUNCTION_8BIT>>4;
64 lcd_e_toggle();
65 zrtos_gpio__delay_microseconds(LCD_DELAY_INIT); /* delay, busy flag can't be checked here */
66
67 /* repeat last command */
68 lcd_e_toggle();
69 zrtos_gpio__delay_microseconds(LCD_DELAY_INIT_REP); /* delay, busy flag can't be checked here */
70
71 /* repeat last command a third time */
72 lcd_e_toggle();
73 zrtos_gpio__delay_microseconds(LCD_DELAY_INIT_REP); /* delay, busy flag can't be checked here */
74
75 /* now configure for 4bit mode */
76 LCD_DATA0_PORT &= ~_BV(LCD_DATA0_PIN); // LCD_FUNCTION_4BIT_1LINE>>4
77 lcd_e_toggle();
78 zrtos_gpio__delay_microseconds(LCD_DELAY_INIT_4BIT); /* some displays need this additional delay */
79
80}

110 ,thiz->pin_data2
111 ,thiz->pin_data3
112 );
113 if ( PIN(LCD_DATA0_PORT) & _BV(LCD_DATA0_PIN) ) data |= 0x10;
114 if ( PIN(LCD_DATA1_PORT) & _BV(LCD_DATA1_PIN) ) data |= 0x20;
115 if ( PIN(LCD_DATA2_PORT) & _BV(LCD_DATA2_PIN) ) data |= 0x40;

123 if ( PIN(LCD_DATA1_PORT) & _BV(LCD_DATA1_PIN) ) data |= 0x20;
124 if ( PIN(LCD_DATA2_PORT) & _BV(LCD_DATA2_PIN) ) data |= 0x40;
125 if ( PIN(LCD_DATA3_PORT) & _BV(LCD_DATA3_PIN) ) data |= 0x80;
126 lcd_e_low();
127
128 return data;

131static bool zrtos_driver_dht11__on_change(void *data){
132 zrtos_driver_dht11_t *thiz = data;
133
134 int val = zrtos_gpio__get_value(thiz->gpio);
135 if(val > 0){
136 thiz->ts = zrtos_time__get_microseconds();
137 }else if(val == 0){
138 if(zrtos_time__get_microseconds() - thiz->ts > 40){
139 zrtos_bitfield__set(thiz->bits,thiz->idx,true);
140 }
141
142 if(++thiz->idx == 40){
143 thiz->error = (bits[0] + bits[2] == bits[4])
144 ? 0
145 : EIO
146 ;
147 goto L_REMOVE_INTERUPT;
148 }
149 }else if(val < 0){
150 thiz->error = -val;
151L_REMOVE_INTERUPT:
152 zrtos_gpio__disable_interrupt(&thiz->gpio);
153 }
154}
