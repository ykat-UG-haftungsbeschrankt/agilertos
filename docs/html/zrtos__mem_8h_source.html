17/**
18 * Swap \p len bytes of \p dest and \p src inplace.
19 *
20 * @param[in] dest destination
21 * @param[in] src source
22 * @param[in] len byte length to swap
23 *
24 * @return Nothing

26void zrtos_mem__swap(void *dest, void *src, size_t len){
27 uint8_t *tmp_dest = dest;
28 uint8_t *tmp_src = src;
29 while(len--){
30 ZRTOS_TYPES__SWAP_PTR_CONTENTS(tmp_src,tmp_dest);
31 tmp_src++;
32 tmp_dest++;
33 }

49void zrtos_mem__left_rotate(
50 void *ptr
51 ,size_t length
52 ,size_t buffer_length
53){
54 size_t d = length;
55 size_t n = buffer_length;
56 size_t i = d;
57 size_t j = n - d;
58 while(i != j){
59 if(i < j){
60 //A is shorter
61 zrtos_mem__swap_ex(ptr, d - i, d + j - i, i);
62 j -= i;
63 }else{
64 //B is shorter
65 zrtos_mem__swap_ex(ptr, d - i, d, j);
66 i -= j;
67 }
68 }
69 //Finally, block swap A and B
70 zrtos_mem__swap_ex(ptr, d - i, d, i);
71}

149void zrtos_mem__sort(void *base,
150 size_t nel,
151 size_t width,
152 int (*comp)(const void *, const void *))
153{
154 size_t wnel, gap, wgap, i, j, k;
155 char *a, *b, tmp;
156
157 wnel = width * nel;
158 for (gap = 0; ++gap < nel;)
159 gap *= 3;
160 while ((gap /= 3) != 0) {
161 wgap = width * gap;
162 for (i = wgap; i < wnel; i += width) {
163 for (j = i - wgap; ;j -= wgap) {
164 a = j + (char *)base;
165 b = a + wgap;
166 if ((*comp)(a, b) <= 0)
167 break;
168 k = width;
169 do {
170 tmp = *a;
171 *a++ = *b;
172 *b++ = tmp;
173 } while (--k);
174 if (j < wgap)
175 break;
176 }
177 }
178 }
