191int pthread_join(pthread_t thread, void **retval){
192 int ret = EAGAIN;
193 zrtos_task_t *task = thread.task;
194
195 while(true){
196 ZRTOS_TASK_SCHEDULER__DO_NOT_DISTURB({
197 if(zrtos_task_scheduler__has_task(task)){
198 if(zrtos_task__is_done(task)){
199 zrtos_task_pthread__trampoline_cb_args_t *args =
200 (zrtos_task_pthread__trampoline_cb_args_t *)(task+1)
201 ;
202 if(retval){
203 *retval = args->return_value;
204 }
205 zrtos_task_pthread__free(task);
206 ret = 0;
207 }
208 }else{
209 ret = ESRCH;
210 }
211 });
212
213 if(ret == EAGAIN){
214 _zrtos_task_scheduler__on_tick_ex();
215 }else{
216 break;
217 }
218 }
219
220 return ret;
221}
