24/**
25 * The card's layout is harddisk-like, which means it contains
26 * a master boot record with a partition table.
27 */
28#define SD_RAW_FORMAT_HARDDISK 0
29/**
30 * The card contains a single filesystem and no partition table.
31 */
32#define SD_RAW_FORMAT_SUPERFLOPPY 1
33/**
34 * The card's layout follows the Universal File Format.
35 */
36#define SD_RAW_FORMAT_UNIVERSAL 2
37/**
38 * The card's layout is unknown.
39 */
40#define SD_RAW_FORMAT_UNKNOWN 3
41
42/**
43 * This struct is used by sd_raw_get_info() to return
44 * manufacturing and status information of the card.
45 */
46struct sd_raw_info
47{
48 /**
49 * A manufacturer code globally assigned by the SD card organization.
50 */
51 unsigned char manufacturer;
52 /**
53 * A string describing the card's OEM or content, globally assigned by the SD card organization.
54 */
55 unsigned char oem[3];
56 /**
57 * A product name.
58 */
59 unsigned char product[6];
60 /**
61 * The card's revision, coded in packed BCD.
62 *
63 * For example, the revision value \c 0x32 means "3.2".
64 */
65 unsigned char revision;
66 /**
67 * A serial number assigned by the manufacturer.
68 */
69 unsigned int serial;
70 /**
71 * The year of manufacturing.
72 *
73 * A value of zero means year 2000.
74 */
75 unsigned char manufacturing_year;
76 /**
77 * The month of manufacturing.
78 */
79 unsigned char manufacturing_month;
80 /**
81 * The card's total capacity in bytes.
82 */
83 unsigned int capacity;
84 /**
85 * Defines wether the card's content is original or copied.
86 *
87 * A value of \c 0 means original, \c 1 means copied.
88 */
89 unsigned char flag_copy;
90 /**
91 * Defines wether the card's content is write-protected.
92 *
93 * \note This is an internal flag and does not represent the
94 * state of the card's mechanical write-protect switch.
95 */
96 unsigned char flag_write_protect;
97 /**
98 * Defines wether the card's content is temporarily write-protected.
99 *
100 * \note This is an internal flag and does not represent the
101 * state of the card's mechanical write-protect switch.
102 */
103 unsigned char flag_write_protect_temp;
104 /**
105 * The card's data layout.
106 *
107 * See the \c SD_RAW_FORMAT_* constants for details.
108 *
109 * \note This value is not guaranteed to match reality.
110 */
111 unsigned char format;
112};
