39/* commands available in SPI mode */
40
41/* CMD0: response R1 */
42#define CMD_GO_IDLE_STATE 0x00
43/* CMD1: response R1 */
44#define CMD_SEND_OP_COND 0x01
45/* CMD9: response R1 */
46#define CMD_SEND_CSD 0x09
47/* CMD10: response R1 */
48#define CMD_SEND_CID 0x0a
49/* CMD12: response R1 */
50#define CMD_STOP_TRANSMISSION 0x0c
51/* CMD13: response R2 */
52#define CMD_SEND_STATUS 0x0d
53/* CMD16: arg0[31:0]: block length, response R1 */
54#define CMD_SET_BLOCKLEN 0x10
55/* CMD17: arg0[31:0]: data address, response R1 */
56#define CMD_READ_SINGLE_BLOCK 0x11
57/* CMD18: arg0[31:0]: data address, response R1 */
58#define CMD_READ_MULTIPLE_BLOCK 0x12
59/* CMD24: arg0[31:0]: data address, response R1 */
60#define CMD_WRITE_SINGLE_BLOCK 0x18
61/* CMD25: arg0[31:0]: data address, response R1 */
62#define CMD_WRITE_MULTIPLE_BLOCK 0x19
63/* CMD27: response R1 */
64#define CMD_PROGRAM_CSD 0x1b
65/* CMD28: arg0[31:0]: data address, response R1b */
66#define CMD_SET_WRITE_PROT 0x1c
67/* CMD29: arg0[31:0]: data address, response R1b */
68#define CMD_CLR_WRITE_PROT 0x1d
69/* CMD30: arg0[31:0]: write protect data address, response R1 */
70#define CMD_SEND_WRITE_PROT 0x1e
71/* CMD32: arg0[31:0]: data address, response R1 */
72#define CMD_TAG_SECTOR_START 0x20
73/* CMD33: arg0[31:0]: data address, response R1 */
74#define CMD_TAG_SECTOR_END 0x21
75/* CMD34: arg0[31:0]: data address, response R1 */
76#define CMD_UNTAG_SECTOR 0x22
77/* CMD35: arg0[31:0]: data address, response R1 */
78#define CMD_TAG_ERASE_GROUP_START 0x23
79/* CMD36: arg0[31:0]: data address, response R1 */

81/* CMD37: arg0[31:0]: data address, response R1 */
82#define CMD_UNTAG_ERASE_GROUP 0x25

92/* command responses */
93/* R1: size 1 byte */
94#define R1_IDLE_STATE 0
95#define R1_ERASE_RESET 1
96#define R1_ILL_COMMAND 2
97#define R1_COM_CRC_ERR 3
98#define R1_ERASE_SEQ_ERR 4
99#define R1_ADDR_ERR 5
100#define R1_PARAM_ERR 6
101/* R1b: equals R1, additional busy bytes */
102/* R2: size 2 bytes */
103#define R2_CARD_LOCKED 0
104#define R2_WP_ERASE_SKIP 1
105#define R2_ERR 2
106#define R2_CARD_ERR 3
107#define R2_CARD_ECC_FAIL 4
108#define R2_WP_VIOLATION 5
109#define R2_INVAL_ERASE 6
110#define R2_OUT_OF_RANGE 7
111#define R2_CSD_OVERWRITE 7
112#define R2_IDLE_STATE (R1_IDLE_STATE + 8)
113#define R2_ERASE_RESET (R1_ERASE_RESET + 8)
114#define R2_ILL_COMMAND (R1_ILL_COMMAND + 8)
115#define R2_COM_CRC_ERR (R1_COM_CRC_ERR + 8)
116#define R2_ERASE_SEQ_ERR (R1_ERASE_SEQ_ERR + 8)
117#define R2_ADDR_ERR (R1_ADDR_ERR + 8)
118#define R2_PARAM_ERR (R1_PARAM_ERR + 8)
119/* R3: size 5 bytes */
120#define R3_OCR_MASK (0xffffffffUL)
121#define R3_IDLE_STATE (R1_IDLE_STATE + 32)
122#define R3_ERASE_RESET (R1_ERASE_RESET + 32)
123#define R3_ILL_COMMAND (R1_ILL_COMMAND + 32)
124#define R3_COM_CRC_ERR (R1_COM_CRC_ERR + 32)

153/**
154 * \ingroup sd_raw
155 * Initializes memory card communication.
156 *
157 * \returns 0 on failure, 1 on success.
158 */
159unsigned char sd_raw_init()
160{
161 /* enable inputs for reading card status */
162 /* configure_pin_available();*/
163 /* configure_pin_locked();*/
164
165 /* enable outputs for MOSI, SCK, SS, input for MISO */
166 configure_pin_ss();
167 configure_pin_mosi();
168 configure_pin_miso();
169 configure_pin_sck();
170
171 unselect_card();
172
173 /* initialize SPI with lowest frequency; max. 400kHz during identification mode of card */
174 S0SPCCR = 150; /* Set frequency to 400kHz */
175 S0SPCR = 0x38;
176
177
178 /* initialization procedure */
179
180 if(!sd_raw_available()) {
181 return 0;
182 }
183 configure_pin_ss();
184 unselect_card();
185
186 unsigned short i;
187 /* card needs 74 cycles minimum to start up */
188 for(i = 0; i < 10; ++i) {
189 /* wait 8 clock cycles */
190 sd_raw_rec_byte();
191 }
192
193 /* address card */
194 select_card();
195
196 /* reset card */
197 unsigned char response;
198 for(i = 0; ; ++i)
199 {
200 response = sd_raw_send_command_r1(CMD_GO_IDLE_STATE, 0);
201 if(response == (1 << R1_IDLE_STATE))
202 break;
203
204 if(i == 0x1ff) {
205 unselect_card();
206 return 0;
207 }
208 }
209
210 /* wait for card to get ready */
211 for(i = 0; ; ++i)
212 {
213 response = sd_raw_send_command_r1(CMD_SEND_OP_COND, 0);
214 if(!(response & (1 << R1_IDLE_STATE)))
215 break;
216
217 if(i == 0x7fff) {
218 unselect_card();
219 return 0;
220 }
221 }
222
223 /* set block size to 512 bytes */
224 if(sd_raw_send_command_r1(CMD_SET_BLOCKLEN, 512)) {
225 unselect_card();
226 return 0;
227 }
228
229 /* deaddress card */
230 unselect_card();
231
232 /* switch to highest SPI frequency possible */
233 S0SPCCR = 10; /* ~1MHz-- potentially can be faster */
234
235 #if !SD_RAW_SAVE_RAM
236 /* the first block is likely to be accessed first, so precache it here */
237 raw_block_address = 0xffffffff;
238 #if SD_RAW_WRITE_BUFFERING
239 raw_block_written = 1;
240 #endif
241 if(!sd_raw_read(0, raw_block, sizeof(raw_block))) {
242 return 0;
243 }
244 #endif
245
246 return 1;
247}

276/**
277 * \ingroup sd_raw
278 * Sends a raw byte to the memory card.
279 *
280 * \param[in] b The byte to sent.
281 * \see sd_raw_rec_byte
282 */
283void sd_raw_send_byte(unsigned char b)
284{
285 S0SPDR = b;
286 /* wait for byte to be shifted out */
287 while(!(S0SPSR & 0x80));
288}

290/**
291 * \ingroup sd_raw
292 * Receives a raw byte from the memory card.
293 *
294 * \returns The byte which should be read.
295 * \see sd_raw_send_byte
296 */
297unsigned char sd_raw_rec_byte(void)
298{
299 /* send dummy data for receiving some */
300 S0SPDR = 0xff;
301 while(!(S0SPSR & 0x80));
302
303 return S0SPDR;
304}

306/**
307 * \ingroup sd_raw
308 * Send a command to the memory card which responses with a R1 response.
309 *
310 * \param[in] command The command to send.
311 * \param[in] arg The argument for command.
312 * \returns The command answer.
313 */
314unsigned char sd_raw_send_command_r1(unsigned char command, unsigned int arg)
315{
316 unsigned char response;
317 unsigned char i;
318
319 /* wait some clock cycles */
320 sd_raw_rec_byte();
321
322 /* send command via SPI */
323 sd_raw_send_byte(0x40 | command);
324 sd_raw_send_byte((arg >> 24) & 0xff);
325 sd_raw_send_byte((arg >> 16) & 0xff);
326 sd_raw_send_byte((arg >> 8) & 0xff);
327 sd_raw_send_byte((arg >> 0) & 0xff);
328 sd_raw_send_byte((command == CMD_GO_IDLE_STATE) ? 0x95 : 0xff);
329
330 /* receive response */
331 for(i = 0; i < 10; ++i)
332 {
333 response = sd_raw_rec_byte();
334 if(response != 0xff)
335 break;
336 }
337
338 return response;
339}

380/**
381 * \ingroup sd_raw
382 * Reads raw data from the card.
383 *
384 * \param[in] offset The offset from which to read.
385 * \param[out] buffer The buffer into which to write the data.
386 * \param[in] length The number of bytes to read.
387 * \returns 0 on failure, 1 on success.
388 * \see sd_raw_read_interval, sd_raw_write
389 */
390unsigned char sd_raw_read(unsigned int offset, unsigned char* buffer, unsigned short length)
391{
392 unsigned int block_address;
393 unsigned short block_offset;
394 unsigned short read_length;
395 while(length > 0)
396 {
397 /* determine byte count to read at once */
398 block_address = offset & 0xfffffe00;
399 block_offset = offset & 0x01ff;
400 read_length = 512 - block_offset; /* read up to block border */
401 if(read_length > length)
402 read_length = length;
403
404 #if !SD_RAW_SAVE_RAM
405 /* check if the requested data is cached */
406 if(block_address != raw_block_address)
407 #endif
408 {
409 #if SD_RAW_WRITE_BUFFERING
410 if(!raw_block_written)
411 {
412 if(!sd_raw_write(raw_block_address, raw_block, sizeof(raw_block)))
413 return 0;
414 }
415 #endif
416
417 /* address card */
418 select_card();
419
420 /* send single block request */
421 if(sd_raw_send_command_r1(CMD_READ_SINGLE_BLOCK, block_address))
422 {
423 unselect_card();
424 return 0;
425 }
426
427 /* wait for data block (start byte 0xfe) */
428 while(sd_raw_rec_byte() != 0xfe);
429
430 #if SD_RAW_SAVE_RAM
431 /* read byte block */
432 unsigned short read_to = block_offset + read_length;
433 for(unsigned short i = 0; i < 512; ++i)
434 {
435 unsigned char b = sd_raw_rec_byte();
436 if(i >= block_offset && i < read_to)
437 *buffer++ = b;
438 }
439 #else
440 /* read byte block */
441 unsigned char* cache = raw_block;
442 unsigned short i;
443 for(i = 0; i < 512; ++i)
444 *cache++ = sd_raw_rec_byte();
445 raw_block_address = block_address;
446
447 memcpy(buffer, raw_block + block_offset, read_length);
448 buffer += read_length;
449 #endif
450
451 /* read crc16 */
452 sd_raw_rec_byte();
453 sd_raw_rec_byte();
454
455 /* deaddress card */
456 unselect_card();
457
458 /* let card some time to finish */
459 sd_raw_rec_byte();
460 }
461 #if !SD_RAW_SAVE_RAM
462 else
463 {
464 /* use cached data */
465 memcpy(buffer, raw_block + block_offset, read_length);
466 }
467 #endif
468
469 length -= read_length;
470 offset += read_length;
471 }
472
473 return 1;
474}

476/**
477 * \ingroup sd_raw
478 * Continuously reads units of \c interval bytes and calls a callback function.
479 *
480 * This function starts reading at the specified offset. Every \c interval bytes,
481 * it calls the callback function with the associated data buffer.
482 *
483 * By returning zero, the callback may stop reading.
484 *
485 * \note Within the callback function, you can not start another read or
486 * write operation.
487 * \note This function only works if the following conditions are met:
488 * - (offset - (offset % 512)) % interval == 0
489 * - length % interval == 0
490 *
491 * \param[in] offset Offset from which to start reading.
492 * \param[in] buffer Pointer to a buffer which is at least interval bytes in size.
493 * \param[in] interval Number of bytes to read before calling the callback function.
494 * \param[in] length Number of bytes to read altogether.
495 * \param[in] callback The function to call every interval bytes.
496 * \param[in] p An opaque pointer directly passed to the callback function.
497 * \returns 0 on failure, 1 on success
498 * \see sd_raw_read, sd_raw_write
499 */
500unsigned char sd_raw_read_interval(unsigned int offset, unsigned char* buffer, unsigned short interval, unsigned short length, sd_raw_interval_handler callback, void* p)
501{
502 if(!buffer || interval == 0 || length < interval || !callback)
503 return 0;
504
505 #if !SD_RAW_SAVE_RAM
506 while(length >= interval)
507 {
508 /* as reading is now buffered, we directly
509 * hand over the request to sd_raw_read()
510 */
511 if(!sd_raw_read(offset, buffer, interval))
512 return 0;
513 if(!callback(buffer, offset, p))
514 break;
515 offset += interval;
516 length -= interval;
517 }
518
519 return 1;
520 #else
521 /* address card */
522 select_card();
523
524 unsigned short block_offset;
525 unsigned short read_length;
526 unsigned char* buffer_cur;
527 unsigned char finished = 0;
528 do
529 {
530 /* determine byte count to read at once */
531 block_offset = offset & 0x01ff;
532 read_length = 512 - block_offset;
533
534 /* send single block request */
535 if(sd_raw_send_command_r1(CMD_READ_SINGLE_BLOCK, offset & 0xfffffe00))
536 {
537 unselect_card();
538 return 0;
539 }
540
541 /* wait for data block (start byte 0xfe) */
542 while(sd_raw_rec_byte() != 0xfe);
543 unsigned short i;
544 /* read up to the data of interest */
545 for(i = 0; i < block_offset; ++i)
546 sd_raw_rec_byte();
547
548 /* read interval bytes of data and execute the callback */
549 do
550 {
551 if(read_length < interval || length < interval)
552 break;
553
554 buffer_cur = buffer;
555 for(i = 0; i < interval; ++i)
556 *buffer_cur++ = sd_raw_rec_byte();
557
558 if(!callback(buffer, offset + (512 - read_length), p))
559 {
560 finished = 1;
561 break;
562 }
563
564 read_length -= interval;
565 length -= interval;
566
567 }
568 while(read_length > 0 && length > 0);
569
570 /* read rest of data block */
571 while(read_length-- > 0)
572 sd_raw_rec_byte();
573
574 /* read crc16 */
575 sd_raw_rec_byte();
576 sd_raw_rec_byte();
577
578 if(length < interval)
579 break;
580
581 offset = (offset & 0xfffffe00) + 512;
582
583 }
584 while(!finished);
585
586 /* deaddress card */
587 unselect_card();
588
589 /* let card some time to finish */
590 sd_raw_rec_byte();
591
592 return 1;
593 #endif
594}

596/**
597 * \ingroup sd_raw
598 * Writes raw data to the card.
599 *
600 * \note If write buffering is enabled, you might have to
601 * call sd_raw_sync() before disconnecting the card
602 * to ensure all remaining data has been written.
603 *
604 * \param[in] offset The offset where to start writing.
605 * \param[in] buffer The buffer containing the data to be written.
606 * \param[in] length The number of bytes to write.
607 * \returns 0 on failure, 1 on success.
608 * \see sd_raw_read
609 */
610unsigned char sd_raw_write(unsigned int offset, const unsigned char* buffer, unsigned short length) {
611#if SD_RAW_WRITE_SUPPORT
612
613 if(get_pin_locked()) return 0;
614
615 unsigned int block_address;
616 unsigned short block_offset;
617 unsigned short write_length;
618 while(length > 0) {
619 /* determine byte count to write at once */
620 block_address = offset & 0xfffffe00;
621 block_offset = offset & 0x01ff;
622 write_length = 512 - block_offset; /* write up to block border */
623 if(write_length > length) write_length = length;
624
625 /* Merge the data to write with the content of the block.
626 * Use the cached block if available.
627 */
628 if(block_address != raw_block_address) {
629#if SD_RAW_WRITE_BUFFERING
630 if(!raw_block_written) {
631 if(!sd_raw_write(raw_block_address, raw_block, sizeof(raw_block))) return 0;
632 }
633#endif
634 if(block_offset || write_length < 512) {
635 if(!sd_raw_read(block_address, raw_block, sizeof(raw_block))) return 0;
636 }
637 raw_block_address = block_address;
638 }
639
640 if(buffer != raw_block) {
641 memcpy(raw_block + block_offset, buffer, write_length);
642
643#if SD_RAW_WRITE_BUFFERING
644 raw_block_written = 0;
645 if(length == write_length) return 1;
646#endif
647 }
648
649 buffer += write_length;
650
651 /* address card */
652 select_card();
653
654 /* send single block request */
655 if(sd_raw_send_command_r1(CMD_WRITE_SINGLE_BLOCK, block_address)) {
656 unselect_card();
657 return 0;
658 }
659
660 /* send start byte */
661 sd_raw_send_byte(0xfe);
662
663 /* write byte block */
664 unsigned char* cache = raw_block;
665 unsigned short i;
666 for(i = 0; i < 512; ++i) sd_raw_send_byte(*cache++);
667
668 /* write dummy crc16 */
669 sd_raw_send_byte(0xff);
670 sd_raw_send_byte(0xff);
671
672 /* wait while card is busy */
673 while(sd_raw_rec_byte() != 0xff);
674 sd_raw_rec_byte();
675
676 /* deaddress card */
677 unselect_card();
678
679 length -= write_length;
680 offset += write_length;
681
682#if SD_RAW_WRITE_BUFFERING
683 raw_block_written = 1;
684#endif
685 }
686
687 return 1;
688#else
689 return 0;
690#endif
691}

720/**
721 * \ingroup sd_raw
722 * Reads informational data from the card.
723 *
724 * This function reads and returns the card's registers
725 * containing manufacturing and status information.
726 *
727 * \note: The information retrieved by this function is
728 * not required in any way to operate on the card,
729 * but it might be nice to display some of the data
730 * to the user.
731 *
732 * \param[in] info A pointer to the structure into which to save the information.
733 * \returns 0 on failure, 1 on success.
734 */
735unsigned char sd_raw_get_info(struct sd_raw_info* info)
736{
737 if(!info || !sd_raw_available())
738 return 0;
739
740 memset(info, 0, sizeof(*info));
741
742 select_card();
743
744 /* read cid register */
745 if(sd_raw_send_command_r1(CMD_SEND_CID, 0))
746 {
747 unselect_card();
748 return 0;
749 }
750 while(sd_raw_rec_byte() != 0xfe);
751 unsigned char i;
752 for(i = 0; i < 18; ++i)
753 {
754 unsigned char b = sd_raw_rec_byte();
755
756 switch(i)
757 {
758 case 0:
759 info->manufacturer = b;
760 break;
761 case 1:
762 case 2:
763 info->oem[i - 1] = b;
764 break;
765 case 3:
766 case 4:
767 case 5:
768 case 6:
769 case 7:
770 info->product[i - 3] = b;
771 break;
772 case 8:
773 info->revision = b;
774 break;
775 case 9:
776 case 10:
777 case 11:
778 case 12:
779 info->serial |= (unsigned int) b << ((12 - i) * 8);
780 break;
781 case 13:
782 info->manufacturing_year = b << 4;
783 break;
784 case 14:
785 info->manufacturing_year |= b >> 4;
786 info->manufacturing_month = b & 0x0f;
787 break;
788 }
789 }

791 /* read csd register */
792 unsigned char csd_read_bl_len = 0;
793 unsigned char csd_c_size_mult = 0;
794 unsigned short csd_c_size = 0;
795 if(sd_raw_send_command_r1(CMD_SEND_CSD, 0))
796 {
797 unselect_card();
798 return 0;
799 }
800 while(sd_raw_rec_byte() != 0xfe);
801 for(i = 0; i < 18; ++i)
802 {
803 unsigned char b = sd_raw_rec_byte();
804
805 switch(i)
806 {
807 case 5:
808 csd_read_bl_len = b & 0x0f;
809 break;
810 case 6:
811 csd_c_size = (unsigned short) (b & 0x03) << 8;
812 break;
813 case 7:
814 csd_c_size |= b;
815 csd_c_size <<= 2;
816 break;
817 case 8:
818 csd_c_size |= b >> 6;
819 ++csd_c_size;
820 break;
821 case 9:
822 csd_c_size_mult = (b & 0x03) << 1;
823 break;
824 case 10:
825 csd_c_size_mult |= b >> 7;
826
827 info->capacity = (unsigned int) csd_c_size << (csd_c_size_mult + csd_read_bl_len + 2);
828
829 break;
830 case 14:
831 if(b & 0x40)
832 info->flag_copy = 1;
833 if(b & 0x20)
834 info->flag_write_protect = 1;
835 if(b & 0x10)
836 info->flag_write_protect_temp = 1;
837 info->format = (b & 0x0c) >> 2;
838 break;
839 }
840 }
841
842 unselect_card();
843
844 return 1;
845}
