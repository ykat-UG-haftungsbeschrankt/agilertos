99 if(!partition ||
100 #if FAT16_WRITE_SUPPORT

103 0
104 #endif

132/**
133 * \ingroup fat16_fs
134 * Reads and parses the header of a FAT16 filesystem.
135 *
136 * \param[inout] fs The filesystem for which to parse the header.
137 * \returns 0 on failure, 1 on success.
138 */
139uint8_t fat16_read_header(struct fat16_fs_struct* fs) {
140 if(!fs) return 0;
141
142 struct partition_struct* partition = fs->partition;
143 if(!partition) {
144 return 0;
145 }
146
147 /* read fat parameters */
148 uint8_t buffer[25];
149 uint32_t partition_offset = partition->offset * 512;
150
151 if(!partition->device_read(partition_offset + 0x0b, buffer, sizeof(buffer)))
152 return 0;
153
154 uint16_t bytes_per_sector = ((uint16_t) buffer[0x00]) |
155 ((uint16_t) buffer[0x01] << 8);
156 uint8_t sectors_per_cluster = buffer[0x02];
157 uint16_t reserved_sectors = ((uint16_t) buffer[0x03]) |
158 ((uint16_t) buffer[0x04] << 8);
159 uint8_t fat_copies = buffer[0x05];
160 uint16_t max_root_entries = ((uint16_t) buffer[0x06]) |
161 ((uint16_t) buffer[0x07] << 8);
162 uint16_t sector_count_16 = ((uint16_t) buffer[0x08]) |
163 ((uint16_t) buffer[0x09] << 8);
164 uint16_t sectors_per_fat = ((uint16_t) buffer[0x0b]) |
165 ((uint16_t) buffer[0x0c] << 8);
166 uint32_t sector_count = ((uint32_t) buffer[0x15]) |
167 ((uint32_t) buffer[0x16] << 8) |
168 ((uint32_t) buffer[0x17] << 16) |
169 ((uint32_t) buffer[0x18] << 24);
170
171 if(sectors_per_fat == 0)
172 /* this is not a FAT16 */
173 return 0;
174
175 if(sector_count == 0)
176 {
177 if(sector_count_16 == 0)
178 /* illegal volume size */
179 return 0;
180 else
181 sector_count = sector_count_16;
182 }
183
184 /* ensure we really have a FAT16 fs here */
185 uint32_t data_sector_count = sector_count
186 - reserved_sectors
187 - (uint32_t) sectors_per_fat * fat_copies
188 - ((max_root_entries * 32 + bytes_per_sector - 1) / bytes_per_sector);
189 uint32_t data_cluster_count = data_sector_count / sectors_per_cluster;
190 if(data_cluster_count < 4085 || data_cluster_count >= 65525)
191 /* this is not a FAT16 */
192 return 0;
193
194 partition->type = PARTITION_TYPE_FAT16;
195
196 /* fill header information */
197 struct fat16_header_struct* header = &fs->header;
198 memset(header, 0, sizeof(*header));
199
200 header->size = sector_count * bytes_per_sector;
201
202 header->fat_offset = /* jump to partition */
203 partition_offset +
204 /* jump to fat */
205 (uint32_t) reserved_sectors * bytes_per_sector;
206 header->fat_size = (data_cluster_count + 2) * 2;
207
208 header->sector_size = bytes_per_sector;
209 header->cluster_size = (uint32_t) bytes_per_sector * sectors_per_cluster;
210
211 header->root_dir_offset = /* jump to fats */
212 header->fat_offset +
213 /* jump to root directory entries */
214 (uint32_t) fat_copies * sectors_per_fat * bytes_per_sector;
215
216 header->cluster_zero_offset = /* jump to root directory entries */
217 header->root_dir_offset +
218 /* skip root directory entries */
219 (uint32_t) max_root_entries * 32;
220
221 return 1;
222}

224/**
225 * \ingroup fat16_fs
226 * Reads a directory entry of the root directory.
227 *
228 * \param[in] fs Descriptor of file system to use.
229 * \param[in] entry_num The index of the directory entry to read.
230 * \param[out] dir_entry Directory entry descriptor which will get filled.
231 * \returns 0 on failure, 1 on success
232 * \see fat16_read_sub_dir_entry, fat16_read_dir_entry_by_path
233 */
234uint8_t fat16_read_root_dir_entry(const struct fat16_fs_struct* fs, uint16_t entry_num, struct fat16_dir_entry_struct* dir_entry) {
235 if(!fs || !dir_entry) return 0;
236
237 /* we read from the root directory entry */
238 const struct fat16_header_struct* header = &fs->header;
239 device_read_interval_t device_read_interval = fs->partition->device_read_interval;
240 uint8_t buffer[32];
241
242 /* seek to the n-th entry */
243 struct fat16_read_callback_arg arg;
244 memset(&arg, 0, sizeof(arg));
245 arg.entry_num = entry_num;
246 if(!device_read_interval(header->root_dir_offset,
247 buffer,
248 sizeof(buffer),
249 header->cluster_zero_offset - header->root_dir_offset,
250 fat16_dir_entry_seek_callback,
251 &arg) ||
252 arg.entry_offset == 0
253 )
254 return 0;
255
256 /* read entry */
257 memset(dir_entry, 0, sizeof(*dir_entry));
258 if(!device_read_interval(arg.entry_offset,
259 buffer,
260 sizeof(buffer),
261 arg.byte_count,
262 fat16_dir_entry_read_callback,
263 dir_entry))
264 return 0;
265
266 return dir_entry->long_name[0] != '\0' ? 1 : 0;
267}

269/**
270 * \ingroup fat16_fs
271 * Reads a directory entry of a given parent directory.
272 *
273 * \param[in] fs Descriptor of file system to use.
274 * \param[in] entry_num The index of the directory entry to read.
275 * \param[in] parent Directory entry descriptor in which to read directory entry.
276 * \param[out] dir_entry Directory entry descriptor which will get filled.
277 * \returns 0 on failure, 1 on success
278 * \see fat16_read_root_dir_entry, fat16_read_dir_entry_by_path
279 */
280uint8_t fat16_read_sub_dir_entry(const struct fat16_fs_struct* fs, uint16_t entry_num, const struct fat16_dir_entry_struct* parent, struct fat16_dir_entry_struct* dir_entry)
281{
282 if(!fs || !parent || !dir_entry)
283 return 0;
284
285 /* we are in a parent directory and want to search within its directory entry table */
286 if(!(parent->attributes & FAT16_ATTRIB_DIR))
287 return 0;
288
289 /* loop through all clusters of the directory */
290 uint8_t buffer[32];
291 uint32_t cluster_offset;
292 uint16_t cluster_size = fs->header.cluster_size;
293 uint16_t cluster_num = parent->cluster;
294 struct fat16_read_callback_arg arg;
295
296 while(1)
297 {
298 /* calculate new cluster offset */
299 cluster_offset = fs->header.cluster_zero_offset + (uint32_t) (cluster_num - 2) * cluster_size;
300
301 /* seek to the n-th entry */
302 memset(&arg, 0, sizeof(arg));
303 arg.entry_num = entry_num;
304 if(!fs->partition->device_read_interval(cluster_offset,
305 buffer,
306 sizeof(buffer),
307 cluster_size,
308 fat16_dir_entry_seek_callback,
309 &arg)
310 )
311 return 0;
312
313 /* check if we found the entry */
314 if(arg.entry_offset)
315 break;
316
317 /* get number of next cluster */
318 if(!(cluster_num = fat16_get_next_cluster(fs, cluster_num)))
319 return 0; /* directory entry not found */
320 }
321
322 memset(dir_entry, 0, sizeof(*dir_entry));
323
324 /* read entry */
325 if(!fs->partition->device_read_interval(arg.entry_offset,
326 buffer,
327 sizeof(buffer),
328 arg.byte_count,
329 fat16_dir_entry_read_callback,
330 dir_entry))
331 return 0;
332
333 return dir_entry->long_name[0] != '\0' ? 1 : 0;
334}

336/**
337 * \ingroup fat16_fs
338 * Callback function for seeking through subdirectory entries.
339 */
340uint8_t fat16_dir_entry_seek_callback(uint8_t* buffer, uint32_t offset, void* p)
341{
342 struct fat16_read_callback_arg* arg = p;
343
344 /* skip deleted or empty entries */
345 if(buffer[0] == FAT16_DIRENTRY_DELETED || !buffer[0])
346 return 1;
347
348 if(arg->entry_cur == arg->entry_num)
349 {
350 arg->entry_offset = offset;
351 arg->byte_count = buffer[11] == 0x0f ?
352 ((buffer[0] & FAT16_DIRENTRY_LFNSEQMASK) + 1) * 32 :
353 32;
354 return 0;
355 }
356
357 /* if we read a 8.3 entry, we reached a new directory entry */
358 if(buffer[11] != 0x0f)
359 ++arg->entry_cur;
360
361 return 1;
362}

364/**
365 * \ingroup fat16_fs
366 * Callback function for reading a directory entry.
367 */
368uint8_t fat16_dir_entry_read_callback(uint8_t* buffer, uint32_t offset, void* p)
369{
370 struct fat16_dir_entry_struct* dir_entry = p;
371
372 /* there should not be any deleted or empty entries */
373 if(buffer[0] == FAT16_DIRENTRY_DELETED || !buffer[0])
374 return 0;
375
376 if(!dir_entry->entry_offset)
377 dir_entry->entry_offset = offset;
378
379 switch(fat16_interpret_dir_entry(dir_entry, buffer))
380 {
381 case 0: /* failure */
382 return 0;
383 case 1: /* buffer successfully parsed, continue */
384 return 1;
385 case 2: /* directory entry complete, finish */
386 return 0;
387 }
388
389 return 0;
390}

392/**
393 * \ingroup fat16_fs
394 * Interprets a raw directory entry and puts the contained
395 * information into the directory entry.
396 *
397 * For a single file there may exist multiple directory
398 * entries. All except the last one are lfn entries, which
399 * contain parts of the long filename. The last directory
400 * entry is a traditional 8.3 style one. It contains all
401 * other information like size, cluster, date and time.
402 *
403 * \param[in,out] dir_entry The directory entry to fill.
404 * \param[in] raw_entry A pointer to 32 bytes of raw data.
405 * \returns 0 on failure, 1 on success and 2 if the
406 * directory entry is complete.
407 */
408uint8_t fat16_interpret_dir_entry(struct fat16_dir_entry_struct* dir_entry, const uint8_t* raw_entry)
409{
410 if(!dir_entry || !raw_entry || !raw_entry[0])
411 return 0;
412
413 char* long_name = dir_entry->long_name;
414 if(raw_entry[11] == 0x0f)
415 {
416 uint16_t char_offset = ((raw_entry[0] & 0x3f) - 1) * 13;
417
418 if(char_offset + 12 < sizeof(dir_entry->long_name))
419 {
420 /* Lfn supports unicode, but we do not, for now.
421 * So we assume pure ascii and read only every
422 * second byte.
423 */
424 long_name[char_offset + 0] = raw_entry[1];
425 long_name[char_offset + 1] = raw_entry[3];
426 long_name[char_offset + 2] = raw_entry[5];
427 long_name[char_offset + 3] = raw_entry[7];
428 long_name[char_offset + 4] = raw_entry[9];
429 long_name[char_offset + 5] = raw_entry[14];
430 long_name[char_offset + 6] = raw_entry[16];
431 long_name[char_offset + 7] = raw_entry[18];
432 long_name[char_offset + 8] = raw_entry[20];
433 long_name[char_offset + 9] = raw_entry[22];
434 long_name[char_offset + 10] = raw_entry[24];
435 long_name[char_offset + 11] = raw_entry[28];
436 long_name[char_offset + 12] = raw_entry[30];
437 }
438
439 return 1;
440 }
441 else
442 {
443 /* if we do not have a long name, take the short one */
444 if(long_name[0] == '\0')
445 {
446 uint8_t i;
447 for(i = 0; i < 8; ++i)
448 {
449 if(raw_entry[i] == ' ')
450 break;
451 long_name[i] = raw_entry[i];
452 }
453 if(raw_entry[8] != ' ')
454 {
455 long_name[i++] = '.';
456
457 uint8_t j = 8;
458 for(; j < 11; ++j)
459 {
460 if(raw_entry[j] != ' ')
461 {
462 long_name[i++] = raw_entry[j];
463 }
464 else
465 {
466 break;
467 }
468 }
469 }
470
471 long_name[i] = '\0';
472 }
473
474 /* extract properties of file and store them within the structure */
475 dir_entry->attributes = raw_entry[11];
476 dir_entry->cluster = ((uint16_t) raw_entry[26]) |
477 ((uint16_t) raw_entry[27] << 8);
478 dir_entry->file_size = ((uint32_t) raw_entry[28]) |
479 ((uint32_t) raw_entry[29] << 8) |
480 ((uint32_t) raw_entry[30] << 16) |
481 ((uint32_t) raw_entry[31] << 24);
482
483 return 2;
484 }
485}

487/**
488 * \ingroup fat16_file
489 * Retrieves the directory entry of a path.
490 *
491 * The given path may both describe a file or a directory.
492 *
493 * \param[in] fs The FAT16 filesystem on which to search.
494 * \param[in] path The path of which to read the directory entry.
495 * \param[out] dir_entry The directory entry to fill.
496 * \returns 0 on failure, 1 on success.
497 * \see fat16_read_dir
498 */
499uint8_t fat16_get_dir_entry_of_path(struct fat16_fs_struct* fs, const char* path, struct fat16_dir_entry_struct* dir_entry)
500{
501 if(!fs || !path || path[0] == '\0' || !dir_entry)
502 return 0;
503
504 if(path[0] == '/')
505 ++path;
506
507 /* begin with the root directory */
508 memset(dir_entry, 0, sizeof(*dir_entry));
509 dir_entry->attributes = FAT16_ATTRIB_DIR;
510
511 if(path[0] == '\0')
512 return 1;
513
514 while(1) {
515 struct fat16_dir_struct dd;
516 int result=fat16_open_dir(&dd, fs, dir_entry);
517 if(result<0) break;
518
519 /* extract the next hierarchy we will search for */
520 const char* sep_pos = strchr(path, '/');
521 if(!sep_pos)
522 sep_pos = path + strlen(path);
523 uint8_t length_to_sep = sep_pos - path;
524
525 /* read directory entries */
526 while(fat16_read_dir(&dd, dir_entry)) {
527 /* check if we have found the next hierarchy */
528 if((strlen(dir_entry->long_name) != length_to_sep ||
529 strncmp(path, dir_entry->long_name, length_to_sep) != 0))
530 continue;
531
532 fat16_close_dir(&dd);
533
534 if(path[length_to_sep] == '\0')
535 /* we iterated through the whole path and have found the file */
536 return 1;
537
538 if(dir_entry->attributes & FAT16_ATTRIB_DIR) {
539 /* we found a parent directory of the file we are searching for */
540 path = sep_pos + 1;
541 break;
542 }
543
544 /* a parent of the file exists, but not the file itself */
545 return 0;
546 }
547
548 fat16_close_dir(&dd);
549 }
550
551 return 0;
552}

555 * \ingroup fat16_fs
556 * Retrieves the next following cluster of a given cluster.
557 *
558 * Using the filesystem file allocation table, this function returns
559 * the number of the cluster containing the data directly following
560 * the data within the cluster with the given number.
561 *
562 * \param[in] fs The filesystem for which to determine the next cluster.
563 * \param[in] cluster_num The number of the cluster for which to determine its successor.

567{
568 if(!fs || cluster_num < 2)
569 return 0;
570
571 /* read appropriate fat entry */
572 uint8_t fat_entry[2];
573 if(!fs->partition->device_read(fs->header.fat_offset + 2 * cluster_num, fat_entry, 2))
574 return 0;
575
576 /* determine next cluster from fat */
577 cluster_num = ((uint16_t) fat_entry[0]) |
578 ((uint16_t) fat_entry[1] << 8);

582 (cluster_num >= FAT16_CLUSTER_RESERVED_MIN && cluster_num <= FAT16_CLUSTER_RESERVED_MAX) ||

584 //*/
585 return 0;
586
587 return cluster_num;

590/**
591 * \ingroup fat16_fs
592 * Appends a new cluster chain to an existing one.
593 *
594 * Set cluster_num to zero to create a completely new one.
595 *
596 * \param[in] fs The file system on which to operate.
597 * \param[in] cluster_num The cluster to which to append the new chain.
598 * \param[in] count The number of clusters to allocate.
599 * \returns 0 on failure, the number of the first new cluster on success.
600 */
601uint16_t fat16_append_clusters(const struct fat16_fs_struct* fs, uint16_t cluster_num, uint16_t count)
602{
603 #if FAT16_WRITE_SUPPORT
604 if(!fs)
605 return 0;
606
607 device_read_t device_read = fs->partition->device_read;
608 device_write_t device_write = fs->partition->device_write;
609 uint32_t fat_offset = fs->header.fat_offset;
610 uint16_t cluster_max = fs->header.fat_size / 2;
611 uint16_t cluster_next = 0;
612 uint16_t count_left = count;
613 uint8_t buffer[2];
614 uint16_t cluster_new;
615 for(cluster_new = 0; cluster_new < cluster_max; ++cluster_new)
616 {
617 if(!device_read(fat_offset + 2 * cluster_new, buffer, sizeof(buffer)))
618 return 0;
619
620 /* check if this is a free cluster */
621 if(buffer[0] == (FAT16_CLUSTER_FREE & 0xff) &&
622 buffer[1] == ((FAT16_CLUSTER_FREE >> 8) & 0xff))
623 {
624 /* allocate cluster */
625 if(count_left == count)
626 {
627 buffer[0] = FAT16_CLUSTER_LAST_MAX & 0xff;
628 buffer[1] = (FAT16_CLUSTER_LAST_MAX >> 8) & 0xff;
629 }
630 else
631 {
632 buffer[0] = cluster_next & 0xff;
633 buffer[1] = (cluster_next >> 8) & 0xff;
634 }
635
636 if(!device_write(fat_offset + 2 * cluster_new, buffer, sizeof(buffer)))
637 break;
638
639 cluster_next = cluster_new;
640 if(--count_left == 0)
641 break;
642 }
643 }
644
645 do
646 {
647 if(count_left > 0)
648 break;
649
650 /* We allocated a new cluster chain. Now join
651 * it with the existing one.
652 */
653 if(cluster_num >= 2)
654 {
655 buffer[0] = cluster_next & 0xff;
656 buffer[1] = (cluster_next >> 8) & 0xff;
657 if(!device_write(fat_offset + 2 * cluster_num, buffer, sizeof(buffer)))
658 break;
659 }
660
661 return cluster_next;
662
663 }
664 while(0);
665
666 /* No space left on device or writing error.
667 * Free up all clusters already allocated.
668 */
669 fat16_free_clusters(fs, cluster_next);
670
671 return 0;
672 #else
673 return 0;
674 #endif
675}

677/**
678 * \ingroup fat16_fs
679 * Frees a cluster chain, or a part thereof.
680 *
681 * Marks the specified cluster and all clusters which are sequentially
682 * referenced by it as free. They may then be used again for future
683 * file allocations.
684 *
685 * \note If this function is used for freeing just a part of a cluster
686 * chain, the new end of the chain is not correctly terminated
687 * within the FAT. Use fat16_terminate_clusters() instead.
688 *
689 * \param[in] fs The filesystem on which to operate.
690 * \param[in] cluster_num The starting cluster of the chain which to free.
691 * \returns 0 on failure, 1 on success.
692 * \see fat16_terminate_clusters
693 */
694uint8_t fat16_free_clusters(const struct fat16_fs_struct* fs, uint16_t cluster_num)
695{
696 #if FAT16_WRITE_SUPPORT
697 if(!fs || cluster_num < 2)
698 return 0;
699
700 uint32_t fat_offset = fs->header.fat_offset;

818/**
819 * \ingroup fat16_file
820 * Reads data from a file.
821 *
822 * The data requested is read from the current file location.
823 *
824 * \param[in] fd The file handle of the file from which to read.
825 * \param[out] buffer The buffer into which to write.
826 * \param[in] buffer_len The amount of data to read.
827 * \returns The number of bytes read, 0 on end of file, or -1 on failure.
828 * \see fat16_write_file
829 */
830int16_t fat16_read_file(struct fat16_file_struct* fd, uint8_t* buffer, uint16_t buffer_len)
831{
832 /* check arguments */
833 if(!fd || !buffer || buffer_len < 1)
834 return -1;
835
836 /* determine number of bytes to read */
837 if(fd->pos + buffer_len > fd->dir_entry.file_size)
838 buffer_len = fd->dir_entry.file_size - fd->pos;
839 if(buffer_len == 0)
840 return 0;
841
842 uint16_t cluster_size = fd->fs->header.cluster_size;
843 uint16_t cluster_num = fd->pos_cluster;
844 uint16_t buffer_left = buffer_len;
845 uint16_t first_cluster_offset = fd->pos % cluster_size;
846
847 /* find cluster in which to start reading */
848 if(!cluster_num)
849 {
850 cluster_num = fd->dir_entry.cluster;
851
852 if(!cluster_num)
853 {
854 if(!fd->pos)
855 return 0;
856 else
857 return -1;
858 }
859
860 if(fd->pos)
861 {
862 uint32_t pos = fd->pos;
863 while(pos >= cluster_size)
864 {
865 pos -= cluster_size;
866 cluster_num = fat16_get_next_cluster(fd->fs, cluster_num);
867 if(!cluster_num)
868 return -1;
869 }
870 }
871 }
872
873 /* read data */
874 do
875 {
876 /* calculate data size to copy from cluster */
877 uint32_t cluster_offset = fd->fs->header.cluster_zero_offset +
878 (uint32_t) (cluster_num - 2) * cluster_size + first_cluster_offset;
879 uint16_t copy_length = cluster_size - first_cluster_offset;
880 if(copy_length > buffer_left)
881 copy_length = buffer_left;
882
883 /* read data */
884 if(!fd->fs->partition->device_read(cluster_offset, buffer, copy_length))
885 return buffer_len - buffer_left;
886
887 /* calculate new file position */
888 buffer += copy_length;
889 buffer_left -= copy_length;
890 fd->pos += copy_length;
891
892 if(first_cluster_offset + copy_length >= cluster_size)
893 {
894 /* we are on a cluster boundary, so get the next cluster */
895 if((cluster_num = fat16_get_next_cluster(fd->fs, cluster_num)))
896 {
897 first_cluster_offset = 0;
898 }
899 else
900 {
901 fd->pos_cluster = 0;
902 return buffer_len - buffer_left;
903 }
904 }
905
906 fd->pos_cluster = cluster_num;
907
908 }
909 while(buffer_left > 0); /* check if we are done */
910
911 return buffer_len;
912}

921 * \param[in] buffer The buffer from which to read the data to be written.
922 * \param[in] buffer_len The amount of data to write.
923 * \returns The number of bytes written, 0 on disk full, or -1 on failure.
924 * \see fat16_read_file

927 if(fd->magic_cookie!=FD_MAGIC_COOKIE) return -7;

968 }

1000 fd->pos_cluster = 0;
1001 break;
1002 }
1003

1006 }

1008 fd->pos_cluster = cluster_num;
1009
1010 }

1014 if(fd->pos > fd->dir_entry.file_size)

1017

1020 /* write directory entry */

1024 * some data to disk. So we calculate the amount of data
1025 * we wrote to disk and which lies within the old file size.

1029 fd->pos = size_old;
1030 }
1031 }

1033 return buffer_len - buffer_left;

1035 #else
1036 return -6;
1037 #endif

1040/**
1041 * \ingroup fat16_file
1042 * Repositions the read/write file offset.
1043 *
1044 * Changes the file offset where the next call to fat16_read_file()
1045 * or fat16_write_file() starts reading/writing.
1046 *
1047 * If the new offset is beyond the end of the file, fat16_resize_file()
1048 * is implicitly called, i.e. the file is expanded.
1049 *
1050 * The new offset can be given in different ways determined by
1051 * the \c whence parameter:
1052 * - \b FAT16_SEEK_SET: \c *offset is relative to the beginning of the file.
1053 * - \b FAT16_SEEK_CUR: \c *offset is relative to the current file position.
1054 * - \b FAT16_SEEK_END: \c *offset is relative to the end of the file.
1055 *
1056 * The resulting absolute offset is written to the location the \c offset
1057 * parameter points to.
1058 *
1059 * \param[in] fd The file decriptor of the file on which to seek.
1060 * \param[in,out] offset A pointer to the new offset, as affected by the \c whence
1061 * parameter. The function writes the new absolute offset
1062 * to this location before it returns.
1063 * \param[in] whence Affects the way \c offset is interpreted, see above.
1064 * \returns 0 on failure, 1 on success.
1065 */
1066uint8_t fat16_seek_file(struct fat16_file_struct* fd, int32_t* offset, uint8_t whence)
1067{
1068 if(!fd || !offset)
1069 return 0;
1070
1071 uint32_t new_pos = fd->pos;
1072 switch(whence)
1073 {
1074 case FAT16_SEEK_SET:
1075 new_pos = *offset;
1076 break;
1077 case FAT16_SEEK_CUR:
1078 new_pos += *offset;
1079 break;
1080 case FAT16_SEEK_END:
1081 new_pos = fd->dir_entry.file_size + *offset;
1082 break;
1083 default:
1084 return 0;
1085 }
1086
1087 if(new_pos > fd->dir_entry.file_size && !fat16_resize_file(fd, new_pos))
1088 return 0;
1089
1090 fd->pos = new_pos;
1091 fd->pos_cluster = 0;
1092
1093 *offset = new_pos;
1094 return 1;
1095}

1097/**
1098 * \ingroup fat16_file
1099 * Resizes a file to have a specific size.
1100 *
1101 * Enlarges or shrinks the file pointed to by the file descriptor to have
1102 * exactly the specified size.
1103 *
1104 * If the file is truncated, all bytes having an equal or larger offset
1105 * than the given size are lost. If the file is expanded, the additional
1106 * bytes are allocated.
1107 *
1108 * \note Please be aware that this function just allocates or deallocates disk
1109 * space, it does not explicitely clear it. To avoid data leakage, this
1110 * must be done manually.
1111 *
1112 * \param[in] fd The file decriptor of the file which to resize.
1113 * \param[in] size The new size of the file.
1114 * \returns 0 on failure, 1 on success.
1115 */
1116uint8_t fat16_resize_file(struct fat16_file_struct* fd, uint32_t size)
1117{
1118 #if FAT16_WRITE_SUPPORT
1119 if(!fd)
1120 return 0;
1121
1122 uint16_t cluster_num = fd->dir_entry.cluster;
1123 uint16_t cluster_size = fd->fs->header.cluster_size;
1124 uint32_t size_new = size;
1125
1126 do
1127 {
1128 if(cluster_num == 0 && size_new == 0)
1129 /* the file stays empty */
1130 break;
1131
1132 /* seek to the next cluster as long as we need the space */
1133 while(size_new > cluster_size)
1134 {
1135 /* get next cluster of file */
1136 uint16_t cluster_num_next = fat16_get_next_cluster(fd->fs, cluster_num);
1137 if(cluster_num_next)

1139 cluster_num = cluster_num_next;
1140 size_new -= cluster_size;
1141 }
1142 else
1143 {
1144 break;
1145 }
1146 }
1147

1149 {

1151 * it to the existing one, if available.
1152 */

1155 ++cluster_count;
1156 uint16_t cluster_new_chain = fat16_append_clusters(fd->fs, cluster_num, cluster_count);
1157 if(!cluster_new_chain)
1158 return 0;
1159
1160 if(!cluster_num)
1161 {
1162 cluster_num = cluster_new_chain;
1163 fd->dir_entry.cluster = cluster_num;
1164 }
1165 }
1166
1167 /* write new directory entry */
1168 fd->dir_entry.file_size = size;
1169 if(size == 0)
1170 fd->dir_entry.cluster = 0;
1171 if(!fat16_write_dir_entry(fd->fs, &fd->dir_entry))
1172 return 0;
1173
1174 if(size == 0)
1175 {
1176 /* free all clusters of file */
1177 fat16_free_clusters(fd->fs, cluster_num);
1178 }
1179 else if(size_new <= cluster_size)
1180 {
1181 /* free all clusters no longer needed */
1182 fat16_terminate_clusters(fd->fs, cluster_num);
1183 }
1184
1185 }
1186 while(0);
1187
1188 /* correct file position */
1189 if(size < fd->pos)
1190 {
1191 fd->pos = size;
1192 fd->pos_cluster = 0;
1193 }
1194
1195 return 1;
1196 #else
1197 return 0;
1198 #endif
1199}

1504 * \ingroup fat16_file
1505 * Creates a file.
1506 *
1507 * Creates a file and obtains the directory entry of the
1508 * new file. If the file to create already exists, the
1509 * directory entry of the existing file will be returned
1510 * within the dir_entry parameter.
1511 *
1512 * \note The file name is not checked for invalid characters.
1513 *
1514 * \note The generation of the short 8.3 file name is quite
1515 * simple. The first eight characters are used for the filename.
1516 * The extension, if any, is made up of the first three characters
1517 * following the last dot within the long filename. If the
1518 * filename (without the extension) is longer than eight characters,
1519 * the lower byte of the cluster number replaces the last two
1520 * characters to avoid name clashes. In any other case, it is your
1521 * responsibility to avoid name clashes.
1522 *
1523 * \param[in] parent The handle of the directory in which to create the file.
1524 * \param[in] file The name of the file to create.
1525 * \param[out] dir_entry The directory entry to fill for the new file.
1526 * \returns 0 on failure, 1 on success.
1527 * \see fat16_delete_file
1528 */

1530{

1537 {

1547

1563 offset = fs->header.root_dir_offset;
1564 offset_to = fs->header.cluster_zero_offset;
1565 dir_entry_offset = offset;
1566 }
1567

1569 {

1599 }

1601 }
