20zrtos_error_t uart3_init(uint16_t baudrate){
21 /* Set baud rate */
22 if (baudrate & 0x8000) {
23 UART3_STATUS = (1<<U2X3); //Enable 2x speed
24 baudrate &= ~0x8000;
25 }
26 UBRR3H = (uint8_t) (baudrate>>8);

29 /* Enable USART receiver and transmitter and receive complete interrupt */

33#ifdef URSEL3
34 UCSR3C = (1<<URSEL3)|(3<<UCSZ30);
35#else

37#endif
