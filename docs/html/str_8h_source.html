243/* Parse S into tokens separated by characters in DELIM.
244 If S is NULL, the saved pointer in SAVE_PTR is used as
245 the next starting point. For example:
246 char s[] = "-abc-=-def";
247 char *sp;
248 x = strtok_r(s, "-", &sp); // x = "abc", sp = "=-def"
249 x = strtok_r(NULL, "-=", &sp); // x = "def", sp = NULL
250 x = strtok_r(NULL, "=", &sp); // x = NULL
251 // s = "abc\0-def\0"
252*/
253char *zrtos_str__tok_r(char *s, const char *delim, char **save_ptr){
254 char *end;
255 if (s == NULL)
256 s = *save_ptr;
257 if (*s == '\0')
258 {
259 *save_ptr = s;
260 return NULL;
261 }
262 /* Scan leading delimiters. */
263 s += zrtos_str__spn(s, delim);
264 if (*s == '\0')
265 {
266 *save_ptr = s;
267 return NULL;
268 }
269 /* Find the end of the token. */
270 end = s + zrtos_str__cspn(s, delim);
271 if (*end == '\0')
272 {
273 *save_ptr = end;
274 return s;
275 }
276 /* Terminate the token and make *SAVE_PTR point past it. */
277 *end = '\0';
278 *save_ptr = end + 1;
279 return s;
280}
